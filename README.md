const searchInput = document.getElementById('search-input');
const searchResults = document.getElementById('search-results');
const loadingMessage = document.getElementById('loading-message');

let totalGames = 0;
let currentLoading = 0;

function displayResultsSequentially(results) {
    searchResults.innerHTML = '';
    totalGames = results.length;
    currentLoading = 0;

    updateLoadingMessage();

    loadNextGame(results, 0);
}

function loadNextGame(games, index) {
    if (index >= games.length) {
        return; 
    }

    const game = games[index];

    const gameElement = document.createElement('div');
    gameElement.classList.add('game');

    const imageElement = document.createElement('img');
    imageElement.src = game.image;
    imageElement.alt = game.name;
    imageElement.onload = () => {
        currentLoading++;
        updateLoadingMessage(); 
        loadNextGame(games, index + 1); 
    };


    const overlayElement = document.createElement('div');
    overlayElement.classList.add('overlay');
    overlayElement.textContent = game.name;


    gameElement.appendChild(imageElement);
    gameElement.appendChild(overlayElement);


    searchResults.appendChild(gameElement);

    gameElement.addEventListener('click', () => {
        window.location.href = game.url;
    });
}

function updateLoadingMessage() {
    loadingMessage.textContent = `Loading ${currentLoading + 1}/${totalGames}`;
}

function searchGames(query) {
    const results = games.filter(game =>
        game.name.toLowerCase().includes(query.toLowerCase())
    );
    displayResultsSequentially(results);
}

searchInput.addEventListener('input', () => {
    const query = searchInput.value.trim();
    searchGames(query);
});

displayResultsSequentially(games);
